;outport --- colocar valor en el puerto escogido
;a dato 
;d puerto (0-7)

.DATA 1000H

.ORG 0100H
;programa principal
MOV A,05H
MOV D,07H
CALL OUTPORT
HLT

;RUTINA OUTPORT
OUTPORT:
	PUSH PSW 
	MOV A,D
	CPI 00H ;MUESTRA POR PUERTO 00
	JNZ S1
	POP PSW
	OUT 00H
	JMP REGRESAR
S1:   CPI 01H
	JNZ S2H
	POP PSW
	OUT 01H
	JMP REGRESAR
REGRESAR: RET

;HACER RESPALDO DE LA INFORMACION, YA QUE EL SE BORRARAN LOS DATOS
