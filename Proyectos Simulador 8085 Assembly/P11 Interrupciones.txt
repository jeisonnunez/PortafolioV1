;Programa de interrupciones
;Fecha: 10/10/2016

.DATA 1000h
codigo7seg: db  77H,44H,3EH,6EH,4DH,6BH,7BH,46H,7FH,4FH
.ORG 0100h

inicio: 	EI
		MVI A,01h
		MVI A,02h
		MVI A,03h
		JMP inicio
		HLT

;-------------------------------------------
;Rutina que muestra la conversion de 	
;--------------------------------------------
mostrar: 	CALL convert7seg
		MOV A,B
		OUT 00h
		

;-------------------------------------------
;Rutina de conversion de digito decimal a 7seg	
;--------------------------------------------
convert7seg:
	PUSH H; guarda en la pila cualquier valor que tenga HL
	LXI H, codigo7seg; graba la dirección de cod7sg en HL
	ADD L
	MOV L,A
	MOV B,M; mueve lo que hay en M a B, M tiene la direccion de HL
	POP H
	RET

;-------------------------------------------
;Rutina de TRAP	
;--------------------------------------------
.ORG 0024h
		CALL mostrar
		MOV A,B
		OUT 00h
		;Aqui debo colocar el codigo de esta rutina
		RET

;-------------------------------------------
;Rutina de RST 7.5	
;--------------------------------------------
.ORG 003Ch
		CALL mostrar
		;Aqui debo colocar el codigo de esta rutina
		RET
